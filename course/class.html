声明类

function Animal(){
	this.name="大毛";
	this.say=function(){
		alert("啊啊啊啊啊");
	};
	this.sleep=function(){
		alert("zｚＺ");
	};
}

使用类

var animal=new Animal();
animal.say();
animal.sleep();

继承类

function Cat(){
	Animal.call(this);//继承Animal
	this.say=function(){
		alert("喵");
	};
}

使用类

var cat=new Cat();
cat.say();
cat.sleep();
