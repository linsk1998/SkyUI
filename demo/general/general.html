$.inArray(arr,value,column)

查找一个对象在数组的位置

var arr=[
	{'id':0,'name':"小明",'age':18},
	{'id':1,'name':"小红",'age':18},
	{'id':2,'name':"小张",'age':18},
	{'id':3,'name':"静静",'age':18}
];
alert($.inArray(arr,2,'id'));          //2
alert($.inArray(arr,"小小",'name'));   //-1
alert($.inArray(arr,18,'age'));        //0
